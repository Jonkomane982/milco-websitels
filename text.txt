    <script>
        /* 
         * 1. UPDATE THIS URL: 
         * Go to WordPress Media Library, click an image, and copy the "File URL".
         * Paste everything EXCEPT the filename here.
         * Example: https://milcols4.wordpress.com/wp-content/uploads/2026/05/
         */
        const wpMediaUrl = 'https://milcols4.wordpress.com/wp-content/uploads/2026/05/'; 

        const products = [
            { id: 1, name: 'Accessories Baskets', price: 120, cat: 'Crafts', stock: 20, img: 'Assessories Baskets.jpg' },
            { id: 2, name: 'Highland Milk', price: 25, cat: 'Food', stock: 20, img: 'Highland milk.png' },
            { id: 3, name: 'JMM Wine', price: 150, cat: 'Food', stock: 20, img: 'JMM Wine.jpg' },
            { id: 4, name: 'Vinegar', price: 35, cat: 'Food', stock: 20, img: 'Vinegar.jpg' },
            { id: 5, name: 'Basotho Blanket', price: 850, cat: 'Crafts', stock: 20, img: 'basotho blanket.jpg' },
            { id: 6, name: 'Boriba Sauce', price: 45, cat: 'Food', stock: 20, img: 'boriba source.jpg' },
            { id: 7, name: 'Decoration Grass Basin and Pot', price: 200, cat: 'Home', stock: 20, img: 'decoration grass Basin and Pot.jpg' },
            { id: 8, name: 'Decoration Grass Basin', price: 150, cat: 'Home', stock: 20, img: 'decoration grass Basin.jpg' },
            { id: 9, name: 'Goat Milk', price: 30, cat: 'Food', stock: 20, img: 'goad milk.jpeg' },
            { id: 10, name: 'Mokorotlo Hat', price: 180, cat: 'Crafts', stock: 20, img: 'Mokorotlo hat.jpg' },
            { id: 11, name: 'Other Local Products', price: 100, cat: 'Food', stock: 20, img: 'other products.jpg' },
            { id: 12, name: 'Pumpkin Juice', price: 40, cat: 'Food', stock: 20, img: 'pumpkin juice.jpg' },
            { id: 13, name: 'Tholoana Drink', price: 30, cat: 'Food', stock: 20, img: 'tholoana Drink.jpg' },
            { id: 14, name: 'Mokorotlo Traditional', price: 220, cat: 'Crafts', stock: 20, img: 'mokorotho.jpg' }
        ];

        function getImgUrl(imgName) {
            // WordPress renames files: "Highland Milk.png" -> "highland-milk.png"
            let formattedName = imgName.toLowerCase().replace(/ /g, '-');
            return wpMediaUrl + formattedName;
        }

        // This runs if an image fails to load
        function handleImgError(imgElement, productName, productId, originalImg) {
            if (!imgElement.dataset.triedBackup) {
                imgElement.dataset.triedBackup = "true";
                imgElement.src = wpMediaUrl + originalImg.replace(/ /g, '%20');
                return;
            }
            
            imgElement.style.display = 'none';
            const container = imgElement.parentElement;
            container.style.background = '#eee'; // Match the gray in screenshot
            container.style.display = 'flex';
            container.style.alignItems = 'center';
            container.style.justifyContent = 'center';
            container.style.cursor = 'pointer';
            container.style.minHeight = '200px';
            container.onclick = () => triggerManualUpload(productId);
            container.innerHTML = `
                <div style="color:#999; font-size:12px; text-align:center; padding:40px 10px;">
                    Image not found:<br>
                    <strong>${productName}</strong><br>
                    <small style="display:block; margin-top:10px; color:#accent;">(Click to upload image manually)</small>
                </div>
            `;
        }

        let cart = JSON.parse(localStorage.getItem('milco_cart')) || [];
        let notifications = JSON.parse(localStorage.getItem('milco_notifs')) || [{ title: 'Welcome to MILCO!', msg: 'Discover the best of Lesotho.' }];

        function showSection(id) {
            document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
            document.getElementById(id).classList.add('active');
            if (id === 'home') renderRecommendations();
            if (id === 'shop') renderShop();
            if (id === 'cart') renderCart();
            if (id === 'notifications') renderNotifications();
            window.scrollTo(0,0);
        }

        function renderShop() {
            const grid = document.getElementById('shop-grid');
            grid.innerHTML = products.map(p => `
                <div class="card">
                    <div class="img-container" style="min-height:200px; background:#f9f9f9; border-radius:12px; margin-bottom:15px; overflow:hidden;">
                        <img src="${getImgUrl(p.img)}" 
                             alt="${p.name}" 
                             style="width:100%; height:200px; object-fit:cover;"
                             onerror="handleImgError(this, '${p.name}', '${p.img}')">
                    </div>
                    <h3>${p.name}</h3>
                    <p style="color: var(--muted)">${p.cat}</p>
                    <div class="price">M${p.price}</div>
                    <p style="font-size: 12px;">Stock: ${p.stock}</p>
                    <button class="btn btn-primary" onclick="addToCart(${p.id})">Add to Cart</button>
                </div>
            `).join('');
        }

        function renderRecommendations() {
            const grid = document.getElementById('recommendations-grid');
            const recs = products.filter(p => !cart.find(c => c.id === p.id)).slice(0, 3);
            grid.innerHTML = recs.map(p => `
                <div class="card">
                    <div class="img-container" style="min-height:200px; background:#f9f9f9; border-radius:12px; margin-bottom:15px; overflow:hidden;">
                        <img src="${getImgUrl(p.img)}" 
                             alt="${p.name}" 
                             style="width:100%; height:200px; object-fit:cover;"
                             onerror="handleImgError(this, '${p.name}', '${p.img}')">
                    </div>
                    <h3>${p.name}</h3>
                    <div class="price">M${p.price}</div>
                    <button class="btn btn-primary" onclick="addToCart(${p.id})">Quick Add</button>
                </div>
            `).join('');
        }

        function addToCart(id) {
            const p = products.find(prod => prod.id === id);
            const exists = cart.find(item => item.id === id);
            if (exists) exists.qty++;
            else cart.push({ ...p, qty: 1 });
            
            saveCart();
            showToast(`${p.name} added!`);
            
            if (cart.length === 2) {
                addNotification('Special Offer', 'Add one more item to get free delivery!');
            }
        }

        function renderCart() {
            const list = document.getElementById('cart-list');
            if (cart.length === 0) {
                list.innerHTML = "<p>Your cart is empty.</p>";
                return;
            }
            list.innerHTML = cart.map(item => `
                <div class="card" style="display:flex; justify-content:space-between; align-items:center; margin-bottom:10px;">
                    <div style="display:flex; align-items:center; gap:15px;">
                        <img src="${getImgUrl(item.img)}" 
                             style="width:50px; height:50px; object-fit:cover; border-radius:8px;" 
                             onerror="this.src='https://via.placeholder.com/50'">
                        <div>
                            <strong>${item.name}</strong><br>
                            M${item.price}
                        </div>
                    </div>
                    <div style="display:flex; align-items:center; gap:10px;">
                        <button class="btn" onclick="updateQty(${item.id}, -1)">-</button>
                        <span>${item.qty}</span>
                        <button class="btn" onclick="updateQty(${item.id}, 1)">+</button>
                        <button class="btn" style="color:red;" onclick="removeFromCart(${item.id})">×</button>
                    </div>
                </div>
            `).join('');
            
            const total = cart.reduce((sum, item) => sum + (item.price * item.qty), 0);
            document.getElementById('total-price').textContent = `Total: M${total.toFixed(2)}`;
        }

        function updateQty(id, delta) {
            const item = cart.find(c => c.id === id);
            item.qty += delta;
            if (item.qty < 1) removeFromCart(id);
            else saveCart();
            renderCart();
        }

        function removeFromCart(id) {
            cart = cart.filter(c => c.id !== id);
            saveCart();
            renderCart();
        }

        function saveCart() {
            localStorage.setItem('milco_cart', JSON.stringify(cart));
            document.getElementById('cart-badge').textContent = cart.reduce((s, i) => s + i.qty, 0);
        }

        function addNotification(title, msg) {
            notifications.unshift({ title, msg, time: new Date().toLocaleTimeString() });
            localStorage.setItem('milco_notifs', JSON.stringify(notifications));
            document.getElementById('notif-badge').textContent = notifications.length;
        }

        function renderNotifications() {
            document.getElementById('notif-list').innerHTML = notifications.map(n => `
                <div class="card" style="margin-bottom:10px; border-left: 4px solid var(--accent);">
                    <strong>${n.title}</strong>
                    <p style="margin:0; font-size:14px;">${n.msg}</p>
                    <small style="color:var(--muted)">${n.time || 'Recently'}</small>
                </div>
            `).join('');
        }

        function processPayment(e) {
            e.preventDefault();
            const orderId = 'MILCO-' + Math.floor(Math.random() * 100000);
            document.getElementById('order-details').innerHTML = `Order ID: <strong>${orderId}</strong><br>Total: <strong>${document.getElementById('total-price').textContent}</strong>`;
            
            // Visual Stock Reduction simulation
            cart.forEach(item => {
                const p = products.find(prod => prod.id === item.id);
                if (p) p.stock -= item.qty;
            });

            cart = [];
            saveCart();
            addNotification('Order Confirmed', `Order ${orderId} is being prepared for delivery.`);
            document.getElementById('success-modal').style.display = 'flex';
        }

        function closeModal() {
            document.getElementById('success-modal').style.display = 'none';
            showSection('home');
        }

        function showToast(msg) {
            const t = document.getElementById('toast');
            t.textContent = msg;
            t.style.display = 'block';
            setTimeout(() => t.style.display = 'none', 3000);
        }

        // Initialize
        renderRecommendations();
        saveCart();
        document.getElementById('notif-badge').textContent = notifications.length;
    </script>
